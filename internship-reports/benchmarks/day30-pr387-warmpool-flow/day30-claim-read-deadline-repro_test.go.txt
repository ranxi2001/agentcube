func TestReproduceClaimReadOutlivesInternalTimer(t *testing.T) {
	const stalledRead = 2*time.Minute + 2*time.Second

	claim := &extensionsv1alpha1.SandboxClaim{
		TypeMeta: metav1.TypeMeta{
			APIVersion: "extensions.agents.x-k8s.io/v1alpha1",
			Kind:       "SandboxClaim",
		},
		ObjectMeta: metav1.ObjectMeta{Name: "ci-claim", Namespace: "ns-1"},
		Status: extensionsv1alpha1.SandboxClaimStatus{
			SandboxStatus: extensionsv1alpha1.SandboxStatus{Name: "adopted-sandbox"},
		},
	}
	readySandbox := &sandboxv1alpha1.Sandbox{
		TypeMeta: metav1.TypeMeta{
			APIVersion: "agents.x-k8s.io/v1alpha1",
			Kind:       "Sandbox",
		},
		ObjectMeta: metav1.ObjectMeta{Name: "adopted-sandbox", Namespace: "ns-1"},
		Status: sandboxv1alpha1.SandboxStatus{Conditions: []metav1.Condition{{
			Type:   string(sandboxv1alpha1.SandboxConditionReady),
			Status: metav1.ConditionTrue,
		}}},
	}

	claimPath := "/apis/extensions.agents.x-k8s.io/v1alpha1/namespaces/ns-1/sandboxclaims/ci-claim"
	sandboxPath := "/apis/agents.x-k8s.io/v1alpha1/namespaces/ns-1/sandboxes/adopted-sandbox"
	apiServer := httptest.NewServer(http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
		w.Header().Set("Content-Type", "application/json")
		switch r.URL.Path {
		case claimPath:
			timer := time.NewTimer(stalledRead)
			defer timer.Stop()
			select {
			case <-timer.C:
			case <-r.Context().Done():
				return
			}
			require.NoError(t, json.NewEncoder(w).Encode(claim))
		case sandboxPath:
			require.NoError(t, json.NewEncoder(w).Encode(readySandbox))
		default:
			http.NotFound(w, r)
		}
	}))
	defer apiServer.Close()

	dynamicClient, err := dynamic.NewForConfig(&rest.Config{Host: apiServer.URL})
	require.NoError(t, err)

	parentCtx, cancel := context.WithTimeout(context.Background(), 3*time.Minute)
	defer cancel()

	startedAt := time.Now()
	createdSandbox, err := (&Server{}).waitForClaimSandboxReady(parentCtx, dynamicClient, claim)
	elapsed := time.Since(startedAt)

	require.NoError(t, err)
	require.NotNil(t, createdSandbox)
	require.Equal(t, readySandbox.Name, createdSandbox.Name)
	require.GreaterOrEqual(t, elapsed, stalledRead)
	t.Logf("BUG REPRODUCED: function returned success after %s, beyond its 2m internal deadline", elapsed.Round(time.Millisecond))
}

func TestControlParentContextCancelsStalledClaimRead(t *testing.T) {
	requestCanceled := make(chan struct{})
	apiServer := httptest.NewServer(http.HandlerFunc(func(_ http.ResponseWriter, r *http.Request) {
		<-r.Context().Done()
		close(requestCanceled)
	}))
	defer apiServer.Close()

	dynamicClient, err := dynamic.NewForConfig(&rest.Config{Host: apiServer.URL})
	require.NoError(t, err)
	claim := &extensionsv1alpha1.SandboxClaim{
		ObjectMeta: metav1.ObjectMeta{Name: "ci-claim", Namespace: "ns-1"},
	}

	parentCtx, cancel := context.WithTimeout(context.Background(), 100*time.Millisecond)
	defer cancel()

	startedAt := time.Now()
	createdSandbox, err := (&Server{}).waitForClaimSandboxReady(parentCtx, dynamicClient, claim)
	elapsed := time.Since(startedAt)

	require.Nil(t, createdSandbox)
	require.ErrorIs(t, err, context.DeadlineExceeded)
	require.Less(t, elapsed, time.Second)
	select {
	case <-requestCanceled:
	case <-time.After(time.Second):
		t.Fatal("dynamic-client request did not observe parent context cancellation")
	}
	t.Logf("CONTROL PASSED: parent context canceled the same stalled GET after %s", elapsed.Round(time.Millisecond))
}
